Example 1: Basic Matrix Algebra


' Computes v^T w for a vector 
set /dotproduct
    (
      join v w multiply .
      collapse ans add "#" .
    ) .


' matrix-multiply A B computes A^T B 
set /matrix-multiply
    (
      set /i /#1 .
      set /out [ list . ] .
      while ( exist-in i B . )
            (
              do-to-all ( dotproduct /v t /w B:i . ) A .
              mov /ans /res .
              in out ( set i res . ) .
              next i .
              mov /ans /i .
            ) .
      answer out .
    ) .

' Computes A^T 
set /transpose
    (
      set /i /#1 .
      set /out [ list . ] .
      while ( exist-in i A:/#1 . )
            (
              do-to-all ( answer t:i . ) A .
              mov /ans /res .
              in out ( set i res . ) .
              next i .
              mov /ans /i .
            ) .
      answer out .
    ) .


set /A [ list [ list 1 2 . ] [ list 3 4 . ] . ] .
set /B [ list [ list 6 2 . ] [ list 3 4 . ] . ] .

Example 2: Numerial Integration


' takes a function as an argument and return left hand
' riemmann sum over range

set /integrate 
    ( 
      set /delta [ divide [ subtract ub lb . ] n . ] .
      set /x lb .
      set /int 0.0 .
      repeat n
            (
              set /int [ add int [ f /t x . ] . ] .
              set /x [ add x delta . ] .
            ) .
      multiply int delta .
    ) .
