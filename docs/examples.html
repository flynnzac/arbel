Example 1: Basic Matrix Algebra


' Computes v^T w for a vector 
set /dotproduct
    (
      do-to-all mul v w .
      collapse add ans  .
    ) .

' matrix-multiply A B computes A^T B 
set /matrix-multiply
    (
      set /i /t1 .
      set /out [ list . ] .
      while ( exist i B . )
            (
              do-to-all < ( dotproduct /v t1 /w B:i . ) /v > A .
              move /ans /res .
              in out ( set i res . ) .
              next i .
              move /ans /i .
            ) .
      answer out .
    ) .

' Computes A^T 
set /transpose
    (
      set /i /t1 .
      set /out [ list . ] .
      while ( exist i A:/t1 . )
            (
              do-to-all < ( answer t1:i . ) /t1 > A .
              move /ans /res .
              in out ( set i res . ) .
              next i .
              move /ans /i .
            ) .
      answer out .
    ) .

Example 2: Numerial Integration


' takes a function as an argument and return left hand
' riemmann sum over range

set /integrate 
    ( 
      set /delta [ div [ sub ub lb . ] n . ] .
      set /x lb .
      set /int 0.0 .
      repeat n
            (
              set /int [ add int [ f /t x . ] . ] .
              set /x [ add x delta . ] .
            ) .
      mul int delta .
    ) .
